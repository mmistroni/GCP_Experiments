class SeekingalphaMarketSpider(scrapy.Spider):
    name = 'seekingalpha_market'
    allowed_domains = ['seekingalpha.com']
    start_urls = ['https://seekingalpha.com/api/sa/combined/AMZN.xml']

    #rules = (
    #    Rule(LinkExtractor(),
    #    callback='parse_page',
    #    # hook to be called when this Rule generates a Request
    #    process_request='add_errback'),
    #                       )
    def parse(self, response):
        items = response.xpath('//item')
        self.logger.info('=============== PARSING ===')
        for index, item in enumerate(items):
            self.logger.info('Parsing Index:%s', index)
            title = ''.join(item.xpath('title/text()').extract())
            link = ''.join(item.xpath('link/text()').extract())
            self.logger.info('Obtained link:%s', link)
            if bool(True): #if 'article' in link:
                self.logger.info('Now attempting to follow:%s', link)
                #yield {'link': link}
                callbackfun = partial(self.parse_attr, 'AMZN')
                yield scrapy.Request(link , callback=callbackfun)

    def parse_attr(self, ticker,  response):
        content = ''.join(response.xpath("//div[@itemprop='articleBody']/p/text()").extract())
        return {'content': content ,
                'ticker' : ticker}

class SeekingalphaCorporateSpider(scrapy.Spider):
    name = 'seekingalpha_corporate'
    allowed_domains = ['seekingalpha.com']
    start_urls = ['https://seekingalpha.com/market-news/all']

    custom_settings = {
                    'DEPTH_LIMIT': 4    # 4 pages seems to hold all news for one day. overlapping slightly with yesterday
                    }

    def parse(self, response):
        self.logger.info("====== GETTING ALL ITEMS for %s===", response.url)
        items = response.xpath("//li[@class='mc']")
        self.logger.info('=============== PARSING ===')
        for index, item in enumerate(items):
            self.logger.info('Parsing Index:%s', index)
            title = items.xpath("div[@class='media-body']/div[@class='title']/a/text()").extract_first()
            asofdate = ''.join(item.xpath("div[@class='media-body'][1]//span/text()").extract())
            content = ' '.join(item.xpath("div[@class='media-body'][1]/div//li/text()").extract())
            symbol = ''.join(item.xpath("div[@class='media-left'][1]//a/text()").extract())
            yield {'symbol' : symbol, 'title':title, 'asofdate': asofdate, 'content':content}
        self.logger.info("End of items. Now moving to next page...")
        next_page = response.xpath("//div[@id='paging']/ul/li[@class='next']/a/@href").extract_first()
        adj_next_page= 'https://seekingalpha.com' + next_page
        self.logger.info('======= now opening up %s', adj_next_page)
        headers = {'User-Agent': 'Mozilla/5.0 (X11; Linux x86_64; rv:48.0) Gecko/20100101 Firefox/48.0'}
        yield response.follow(next_page, callback=self.parse, headers=headers)
